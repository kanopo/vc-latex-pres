\definecolor{blockTitleFG}{RGB}{4,6,76}
\definecolor{blockTitleBG}{RGB}{252,187,6}
\definecolor{blockBodyFG}{RGB}{0,0,0}
\definecolor{blockBodyBG}{RGB}{255,255,200}

\newenvironment<>{craneblock}[1]{%
	\setbeamercolor{block title}{fg=blockTitleFG, bg=blockTitleBG}%
	\setbeamercolor{block body}{fg=blockBodyFG,bg=blockBodyBG}%
	\begin{block}#2{#1}}{\end{block}}

\newenvironment<>{postitblock}[1]{%
	\setbeamercolor{block title}{fg=blockTitleFG, bg=yellow!80!black}%
	\setbeamercolor{block body}{fg=blockBodyFG, bg=yellow!90!white}%
	\begin{block}#2{#1}}{\end{block}}
